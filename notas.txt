#11g.Dentro de la app_ecommerce crear/modificar el archivo models.py
    from django.db import models

    class Cliente(models.Model):
        nombre = models.CharField(max_length=51)
        apellido = models.CharField(max_length=51)
        tipodoc = models.CharField(max_length=6)
        nrodoc = models.IntegerField(unique=True)
        email = models.EmailField()
        fecha_registro = models.DateTimeField(auto_now_add=True)
        fecha_nacimiento = models.DateField(null=True, blank=True)
        domicilio = models.CharField(max_length=100)

        def __str__(self):
            return f"{self.nombre} {self.apellido}"
